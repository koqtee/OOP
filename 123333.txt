with model_stats as (
select d.dev_model as model, count(rr.rep_id) as req_count, count(d.dev_id) as dev_count,
(count(rr.rep_id) / count(d.dev_id)) as problems
from repair_requests rr 
join devices d on rr.req_device_id = d.dev_id 
group by d.dev_model 
)
select model, req_count, dev_count 
from model_stats


// 2 zadanie //
with 
date_cte as (
	select dcl.replacement_date as date, 
	c.cons_id as id
	from device_consumable_log dcl 
	join consumables c on dcl.consumable_id = c.cons_id 
	order by date
), 
days_cte as (
	select (to_char(dc.date, 'j')::integer) as date_in_days, 
	id 
	from date_cte dc
), 
diff_cte as (
	select (date_in_days - min(date_in_days) over (partition by id)) as days, 
	id 
	from days_cte dac
	order by id, days
	),
max_cte as (
	select (max(days) over (partition by id)) as max_date, 
	id 
	from diff_cte
),
avg_cte as (
	select (max_date / count(id)) as avg, 
	id 
	from max_cte
	group by max_date, id
	order by avg 
)
select * 
from diff_cte

// zadanie 3 // 
with monthly_repairs as (
select
	date_trunc('month', created_at) as repair_month,
	count(rep_id) as repairs_count
from
	repair_requests
group by
	date_trunc('month', created_at)
)
select
	to_char(repair_month, 'YYYY-MM') as month,
	repairs_count,
	round(
        avg(repairs_count) over(
            order by repair_month
            rows between 2 preceding and current row
        ), 2
    ) as rolling_avg
from
	monthly_repairs
	where repair_month >= current_date - interval'4 month'
order by
	repair_month;

create table Departments (
dep_id serial primary key,
dep_name varchar(100) not null,
dep_location varchar(100) not null,
dep_phone varchar(30) not null,
)


create table Employees (
emp_id INT GENERATED BY DEFAULT AS IDENTITY primary key,
emp_full_name varchar(100) not null,
emp_email varchar(100),
emp_position varchar(30) not null,
department_id int,
constraint fk_employees_departments foreign key (department_id) references Departments(dep_id)  
)

create table Devices (
dev_id INT GENERATED BY DEFAULT AS IDENTITY primary key,
inventory_number varchar(100) not null,
dev_type varchar(100) not null,
dev_model varchar(100),
serial_number varchar(100) not null,
purchase_date date, 
warranty_end date, 
status varchar(100) not null,
employee_id int,
constraint fk_devices_employees foreign key (employee_id) references Employees(emp_id)  
)

create table Consumables (
cons_id INT GENERATED BY DEFAULT AS IDENTITY primary key,
cons_name varchar(100) not_null,
compatible_models varchar(250) not null,  
)

create table Device_Consumable_Log (
log_id INT GENERATED BY DEFAULT AS IDENTITY primary key,
replacement_date date,
pages_printed_before int,
device_id int,
consumable_id int,
employee_id int,
constraint fk_log_devices foreign key (device_id) references Devices(dev_id),
constraint fk_log_employees foreign key (employee_id) references Employees(emp_id),
constraint fk_log_consumables foreign key (consumable_id) references Consumables(cons_id)
)

create table Repair_Requests (
rep_id INT GENERATED BY DEFAULT AS IDENTITY primary key,
issue_description varchar(250),
created_at date,
resolved_at date,
status varchar(50) not null,
req_device_id int,
req_employee_id int,
constraint fk_repair_devices foreign key (req_device_id) references Devices(dev_id),
constraint fk_repair_employees foreign key (req_employee_id) references Employess(emp_id),
)
